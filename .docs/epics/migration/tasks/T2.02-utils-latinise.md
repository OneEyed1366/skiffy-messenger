//#region Latinise Functions

/**
 * Maps a single character to its Latin equivalent.
 * Returns the original character if no mapping exists.
 */
export function mapLatinChar(char: string): string {
  return LATIN_MAP[char] ?? char;
}

/**
 * Converts a string by replacing accented/special Latin characters
 * with their ASCII equivalents.
 *
 * @example
 * latinise("Crème brûlée") // "Creme brulee"
 * latinise("Łódź")         // "Lodz"
 * latinise("Ñoño")         // "Nono"
 */
export function latinise(input: string): string {
  return input.replace(/[^A-Za-z0-9]/g, mapLatinChar);
}
